5.9 Таблица перевода расстояний в дюймах в сантиметры 
function task5_9() {
    console.log('Дюймы  | Сантиметры');
    for (let inch = 10; inch <= 22; inch++) {
        let cm = inch * 25.4;
        console.log(`${inch} дюймов  | ${cm.toFixed(2)} см`);
    }
}
5.10 Таблица перевода долларов США в рубли 
function task5_10() {
    let rate = prompt("Введите курс доллара к рублю:");
    for (let dollars = 1; dollars <= 20; dollars++) {
        let rubles = dollars * rate;
        console.log(`${dollars} доллар(ов) = ${rubles.toFixed(2)} рублей`);
    }
}
5.11 Таблица зависимости плотности воздуха от высоты 
function task5_11() {
    let p0 = 1.29;
    let z = 1.25e-4;
    console.log('Высота (м)  | Плотность (кг/м³)');
    for (let h = 0; h <= 1000; h += 100) {
        let p = p0 * Math.exp(-z * h);
        console.log(`${h} | ${p.toFixed(2)}`);
    }
}

5.17 Рассчитать значения y для значений х, равных 4, 5, ..., 28: y = 3t² + 4,87t – 3, t = x + 3 
            function task5_17() {
                console.log('x   | y');
                for (let x = 4; x <= 28; x++) {
                    let t = x + 3;
                    let y = 3 * t * t + 4.87 * t - 3;
                    console.log(`${x}  | ${y.toFixed(2)}`);
                }
            }

5.21 Напечатать таблицу стоимости 50, 100, 150, ..., 1000 г сыра (стоимость 1 кг сыра вводится с клавиатуры) 
            function task5_21() {

                }
            }
